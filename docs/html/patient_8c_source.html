7 * static memory (no heap allocation) in compliance with IEC 62304 Class B
8 * guidance for embedded medical device software.
9 *
10 * ### Memory Safety
11 * - `patient_init()` zero-fills the entire record before writing fields,
12 * preventing stale data from a previous session.
13 * - `strncpy` is used with an explicit length limit and explicit
14 * null-termination to guard against buffer overrun.

31/**
32 * @brief Initialise a PatientRecord, zeroing all fields before populating.
33 * @details memset(0) ensures that any unused readings array slots contain
34 * all-zero VitalSigns, which classify as CRITICAL across all
35 * parameters — a deliberately safe default that prevents silent
36 * display of uninitialised data.
37 */
38void patient_init(PatientRecord *rec, int id, const char *name,
39 int age, float weight_kg, float height_m)
40{
41 memset(rec, 0, sizeof(*rec));
42 rec->info.id = id;
43 rec->info.age = age;
44 rec->info.weight_kg = weight_kg;
45 rec->info.height_m = height_m;
46 strncpy(rec->info.name, name, MAX_NAME_LEN - 1);
47 rec->info.name[MAX_NAME_LEN - 1] = '\0'; /* guarantee null-termination */
48 rec->reading_count = 0;
49}

51/**
52 * @brief Append a vital signs reading, enforcing the MAX_READINGS cap.
53 * @details VitalSigns is copied by value (struct assignment), so the caller
54 * retains ownership of the original and may reuse the variable.
55 */
56int patient_add_reading(PatientRecord *rec, const VitalSigns *v)
57{
58 if (rec->reading_count >= MAX_READINGS) return 0;
59 rec->readings[rec->reading_count++] = *v;
60 return 1;
61}
62
63/**
64 * @brief Return a pointer to the most recent reading, or NULL if none exist.
65 * @details Index arithmetic: the last valid slot is readings[reading_count - 1].
66 */
67const VitalSigns *patient_latest_reading(const PatientRecord *rec)
68{
69 if (rec->reading_count == 0) return NULL;
70 return &rec->readings[rec->reading_count - 1];
71}

73/**
74 * @brief Return the overall status from the latest reading.
75 * @details Delegates entirely to overall_alert_level() to keep classification
76 * logic centralised in vitals.c — consistent with single-responsibility.
77 */
78AlertLevel patient_current_status(const PatientRecord *rec)
79{
80 const VitalSigns *v = patient_latest_reading(rec);
81 if (v == NULL) return ALERT_NORMAL;
82 return overall_alert_level(v);
83}

85/**

88int patient_is_full(const PatientRecord *rec)

93/**
94 * @brief Print a formatted patient summary including vitals and active alerts.
95 * @details Generates alerts inline for the latest reading using generate_alerts().
96 * Output is formatted for an 80-column terminal with box-drawing borders.
97 */
98void patient_print_summary(const PatientRecord *rec)
99{
100 float bmi = calculate_bmi(rec->info.weight_kg, rec->info.height_m);
101 const VitalSigns *latest = patient_latest_reading(rec);
102 AlertLevel status = patient_current_status(rec);
103 Alert alerts[MAX_ALERTS];
104 int alert_count = 0;
105
106 printf("+--------------------------------------------------+\n");
107 printf("| PATIENT SUMMARY |\n");
108 printf("+--------------------------------------------------+\n");
109 printf(" Name : %s\n", rec->info.name);
110 printf(" ID : %d\n", rec->info.id);
111 printf(" Age : %d years\n", rec->info.age);
112 printf(" BMI : %.1f (%s)\n", bmi, bmi_category(bmi));
113 printf(" Readings: %d / %d\n", rec->reading_count, MAX_READINGS);
114
115 if (latest) {
116 printf("\n Latest Vitals:\n");
117 printf(" Heart Rate : %3d bpm [%s]\n",
118 latest->heart_rate,
119 alert_level_str(check_heart_rate(latest->heart_rate)));
120 printf(" BP : %3d/%-3d mmHg [%s]\n",
121 latest->systolic_bp, latest->diastolic_bp,
122 alert_level_str(check_blood_pressure(latest->systolic_bp,
123 latest->diastolic_bp)));
124 printf(" Temperature : %.1f C [%s]\n",
125 latest->temperature,
126 alert_level_str(check_temperature(latest->temperature)));
127 printf(" SpO2 : %3d%% [%s]\n",
128 latest->spo2,
129 alert_level_str(check_spo2(latest->spo2)));
130
131 alert_count = generate_alerts(latest, alerts, MAX_ALERTS);
132 }
133
134 printf("\n Overall Status : %s\n", alert_level_str(status));
135
136 if (alert_count > 0) {
137 printf("\n Active Alerts:\n");
138 for (int i = 0; i < alert_count; i++) {
139 const char *pfx = (alerts[i].level == ALERT_CRITICAL)
140 ? " !! CRITICAL" : " ! WARNING ";
141 printf("%s %s\n", pfx, alerts[i].message);
142 }
143 }
144 printf("+--------------------------------------------------+\n");
145}
