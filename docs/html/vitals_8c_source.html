24/**
25 * @brief Classify a heart rate reading.
26 * @details Implements SWR-VIT-001 threshold logic.
27 * Critical: bpm < 40 (severe bradycardia) or bpm > 150 (severe tachycardia).
28 * Warning: bpm < 60 or bpm > 100.
29 * Normal: 60 <= bpm <= 100.
30 */
31AlertLevel check_heart_rate(int bpm)
32{
33 if (bpm < 40 || bpm > 150) return ALERT_CRITICAL;
34 if (bpm < 60 || bpm > 100) return ALERT_WARNING;
35 return ALERT_NORMAL;
36}

38/**
39 * @brief Classify a blood pressure reading.
40 * @details Implements SWR-VIT-002 threshold logic.
41 * Critical thresholds are evaluated first (fail-safe ordering).
42 * If either systolic or diastolic is critical, CRITICAL is returned.
43 * If either is outside the normal range, WARNING is returned.
44 */
45AlertLevel check_blood_pressure(int systolic, int diastolic)
46{
47 if (systolic < 70 || systolic > 180) return ALERT_CRITICAL;
48 if (diastolic < 40 || diastolic > 120) return ALERT_CRITICAL;
49 if (systolic < 90 || systolic > 140) return ALERT_WARNING;
50 if (diastolic < 60 || diastolic > 90) return ALERT_WARNING;
51 return ALERT_NORMAL;
52}

54/**
55 * @brief Classify a temperature reading.
56 * @details Implements SWR-VIT-003 threshold logic.
57 * <35.0 °C = hypothermia (CRITICAL).
58 * >39.5 °C = hyperpyrexia (CRITICAL).
59 * Values outside 36.1–37.2 but within critical bounds = WARNING.
60 */
61AlertLevel check_temperature(float temp_c)
62{
63 if (temp_c < 35.0f || temp_c > 39.5f) return ALERT_CRITICAL;
64 if (temp_c < 36.1f || temp_c > 37.2f) return ALERT_WARNING;
65 return ALERT_NORMAL;
66}

68/**
69 * @brief Classify an SpO2 reading.
70 * @details Implements SWR-VIT-004 threshold logic.
71 * <90% = hypoxaemia (CRITICAL).
72 * 90–94% = mild hypoxaemia (WARNING).
73 * 95–100% = normal saturation.

75AlertLevel check_spo2(int spo2)

77 if (spo2 < 90) return ALERT_CRITICAL;

82/**
83 * @brief Return the highest alert level across all four vital parameters.
84 * @details Implements SWR-VIT-005. Iterates all four checks and tracks the
85 * maximum. Uses a fixed-size array to avoid branch explosion and
86 * keep cyclomatic complexity low for IEC 62304 review.
87 */
88AlertLevel overall_alert_level(const VitalSigns *vitals)
89{
90 AlertLevel levels[4];
91 AlertLevel max = ALERT_NORMAL;
92 int i;
93
94 levels[0] = check_heart_rate(vitals->heart_rate);
95 levels[1] = check_blood_pressure(vitals->systolic_bp, vitals->diastolic_bp);
96 levels[2] = check_temperature(vitals->temperature);
97 levels[3] = check_spo2(vitals->spo2);
98
99 for (i = 0; i < 4; i++) {
100 if (levels[i] > max) max = levels[i];
101 }
102 return max;
103}

105/**
106 * @brief Calculate BMI using the standard WHO formula.
107 * @details BMI = weight_kg / (height_m^2).
108 * Returns -1.0 for invalid (zero or negative) height to allow callers
109 * to distinguish an error from a legitimate very-low BMI.
110 */
111float calculate_bmi(float weight_kg, float height_m)
112{
113 if (height_m <= 0.0f) return -1.0f;
114 return weight_kg / (height_m * height_m);
115}

117/**
118 * @brief Map a BMI value to a WHO weight-status category string.
119 * @details Uses half-open intervals consistent with WHO classification 2004.
120 */
121const char *bmi_category(float bmi)
122{
123 if (bmi < 0.0f) return "Invalid";
124 if (bmi < 18.5f) return "Underweight";
125 if (bmi < 25.0f) return "Normal weight";
126 if (bmi < 30.0f) return "Overweight";
127 return "Obese";
