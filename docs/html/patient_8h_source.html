1/**
2 * @file patient.h
3 * @brief Patient record management — demographic data and vital sign history.
4 *
5 * @details
6 * This module provides the top-level data structure for a monitored patient
7 * and the functions to manage their vital sign history. It composes the
8 * vitals and alerts modules to provide a complete patient-status API.
9 *
10 * ### Architecture Position
11 * PatientRecord sits at the top of the module hierarchy:
12 * @code
13 * patient.h
14 * └── vitals.h (classification of individual readings)
15 * └── alerts.h (alert record generation)
16 * @endcode
17 *
18 * ### Storage Model
19 * Readings are stored in a fixed-size array (MAX_READINGS = 10). Once full,
20 * further additions are rejected (patient_add_reading() returns 0). This
21 * design avoids dynamic memory allocation, a requirement for safety-critical
22 * embedded software under IEC 62304.
23 *
24 * ### IEC 62304 Traceability
25 * - Software Unit: UNIT-PAT
26 * - Requirements covered: SWR-PAT-001 through SWR-PAT-006
27 *
28 * @version 1.0.0
29 * @date 2026-04-06
30 * @author vinu-engineer
31 *
32 * @copyright Medical Device Software — IEC 62304 Class B compliant.
33 * All rights reserved.
34 */

56/**
57 * @brief Demographic information for a monitored patient.
58 *
59 * @details This structure stores static patient data that does not change
60 * during a monitoring session. It is embedded within PatientRecord.
61 */
62typedef struct {
63 int id; /**< Unique patient identifier. */
64 char name[MAX_NAME_LEN]; /**< Full name. Null-terminated, truncated to
65 MAX_NAME_LEN - 1 characters if necessary. */
66 int age; /**< Age in years. */
67 float weight_kg; /**< Body weight in kilograms. */
68 float height_m; /**< Height in metres. */
69} PatientInfo;
70
71/**
72 * @brief Complete patient record including demographics and vital sign history.
73 *
74 * @details The readings array acts as a sequential log: index 0 holds the
75 * first reading, index reading_count-1 holds the most recent. The record
76 * is considered full when reading_count == MAX_READINGS.
77 *
78 * @note No dynamic memory is used. All storage is stack/static allocated.
79 */
80typedef struct {
81 PatientInfo info; /**< Demographic data. */
82 VitalSigns readings[MAX_READINGS]; /**< Historical vital sign readings. */
83 int reading_count; /**< Number of valid entries in
84 readings[]. Range: 0–MAX_READINGS. */
85} PatientRecord;

97 *

103 * @param[in] height_m Height in metres.
104 *
105 * @pre rec != NULL

108 *
109 * @par Requirement

119/**
120 * @brief Append a vital sign reading to the patient's history.
121 *
122 * @details The VitalSigns structure is copied by value into the internal
123 * array. If the record is already full (reading_count == MAX_READINGS),
124 * the record is left unchanged and 0 is returned.
125 *
126 * @param[in,out] rec Pointer to an initialised PatientRecord. Must not be NULL.
127 * @param[in] v Pointer to the VitalSigns reading to append. Copied
128 * by value; the caller may free or reuse @p v afterwards.
129 *
130 * @return 1 if the reading was successfully appended, 0 if the buffer is full.
131 *
132 * @pre rec != NULL
133 * @pre v != NULL
134 * @post On success: rec->reading_count is incremented by 1.
135 * @post On failure: rec is unchanged.
136 *
137 * @par Requirement
138 * SWR-PAT-002
139 */
140int patient_add_reading(PatientRecord *rec, const VitalSigns *v);
