32/**
33 * @brief Print a single vital sign reading with per-parameter status and alerts.
34 *
35 * @param[in] num Reading sequence number (1-based), used in the heading.
36 * @param[in] v Pointer to the VitalSigns to display. Must not be NULL.
37 */
38static void print_reading(int num, const VitalSigns *v)
39{
40 Alert alerts[MAX_ALERTS];
41 int n, i;
42 AlertLevel status = overall_alert_level(v);
43
44 printf("\n --- Reading #%d ---\n", num);
45 printf(" Heart Rate : %3d bpm [%s]\n",
46 v->heart_rate,
47 alert_level_str(check_heart_rate(v->heart_rate)));
48 printf(" BP : %3d/%-3d mmHg [%s]\n",
49 v->systolic_bp, v->diastolic_bp,
50 alert_level_str(check_blood_pressure(v->systolic_bp, v->diastolic_bp)));
51 printf(" Temperature : %.1f C [%s]\n",
52 v->temperature,
53 alert_level_str(check_temperature(v->temperature)));
54 printf(" SpO2 : %3d%% [%s]\n",
55 v->spo2,
56 alert_level_str(check_spo2(v->spo2)));
57 printf(" Status : %s\n", alert_level_str(status));
58
59 n = generate_alerts(v, alerts, MAX_ALERTS);
60 for (i = 0; i < n; i++) {
61 const char *pfx = (alerts[i].level == ALERT_CRITICAL)
62 ? " !! CRITICAL" : " ! WARNING ";
63 printf("%s %s\n", pfx, alerts[i].message);
64 }
65}

67/**

70 * @details Admits Sarah Johnson and records three readings that progress

73static void monitor_patient(void)

75 PatientRecord rec;

80 printf("==================================================\n");

83 printf("==================================================\n");
84 printf("\n Admitting: %s (ID %d, Age %d)\n",
85 rec.info.name, rec.info.id, rec.info.age);
86 printf(" BMI : %.1f (%s)\n",
87 calculate_bmi(rec.info.weight_kg, rec.info.height_m),
88 bmi_category(calculate_bmi(rec.info.weight_kg, rec.info.height_m)));
89
90 v = (VitalSigns){ 78, 122, 82, 36.7f, 98 }; /* normal admission */
91 patient_add_reading(&rec, &v);
92 print_reading(1, &v);
93
94 v = (VitalSigns){ 108, 148, 94, 37.9f, 93 }; /* mild deterioration */
95 patient_add_reading(&rec, &v);
96 print_reading(2, &v);
97
98 v = (VitalSigns){ 135, 175, 112, 39.8f, 87 }; /* acute crisis */
99 patient_add_reading(&rec, &v);
100 print_reading(3, &v);
101
102 printf("\n");
103 patient_print_summary(&rec);
104 printf("\n");
105}
