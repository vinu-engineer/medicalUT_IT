1/**
2 * @file alerts.h
3 * @brief Alert record generation from a set of vital sign measurements.
4 *
5 * @details
6 * This module is responsible for translating a VitalSigns observation into
7 * a structured list of Alert records — one per abnormal parameter. It acts
8 * as the bridge between raw clinical data (vitals.h) and the patient
9 * management layer (patient.h).
10 *
11 * ### Design Decisions
12 * - Each abnormal parameter produces exactly one Alert record.
13 * - The output buffer size is controlled by the caller via `max_out`, making
14 * the function safe for use in memory-constrained embedded targets.
15 * - All-normal vitals produce zero alerts (return value 0).
16 * - Alert messages are human-readable and suitable for display on a clinical
17 * workstation or serial console.
18 *
19 * ### IEC 62304 Traceability
20 * - Software Unit: UNIT-ALT
21 * - Requirements covered: SWR-ALT-001 through SWR-ALT-004
22 *
23 * @version 1.0.0
24 * @date 2026-04-06
25 * @author vinu-engineer
26 *
27 * @copyright Medical Device Software — IEC 62304 Class B compliant.
28 * All rights reserved.
29 */

69/**
70 * @brief Generate alert records for every vital sign parameter outside
71 * its normal clinical range.
72 *
73 * @details
74 * Iterates over all four vital sign parameters in this fixed order:
75 * -# Heart rate
76 * -# Blood pressure (systolic / diastolic)
77 * -# Temperature
78 * -# SpO2
79 *
80 * For each parameter that is not ALERT_NORMAL, one Alert entry is written
81 * to @p out. Writing stops when either all parameters have been checked or
82 * @p max_out entries have been filled — whichever comes first.
83 *
84 * **Memory safety:** the function never writes beyond `out[max_out - 1]`.
85 * Passing `max_out = 0` is valid and writes nothing.
86 *
87 * @param[in] vitals Pointer to a fully populated VitalSigns structure.
88 * Must not be NULL.
89 * @param[out] out Caller-allocated array to receive Alert records.
90 * Must hold at least @p max_out elements.
91 * @param[in] max_out Maximum number of Alert entries to write.
92 *
93 * @return Number of Alert records written to @p out (0 if all parameters
94 * are normal, or if max_out == 0).
95 *
96 * @pre vitals != NULL
97 * @pre out != NULL || max_out == 0
98 * @post Return value <= max_out
99 *
100 * @par Requirements
101 * SWR-ALT-001, SWR-ALT-002, SWR-ALT-003, SWR-ALT-004
102 */
103int generate_alerts(const VitalSigns *vitals, Alert *out, int max_out);
