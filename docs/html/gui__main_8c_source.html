258/**
259 * @brief Paint one colour-coded vital sign tile.
260 *
261 * Each tile shows the parameter label, measured value + unit, and a
262 * colour-coded badge (NORMAL / WARNING / CRITICAL).
263 *
264 * @req SWR-GUI-003
265 */
266static void paint_tile(HDC hdc,
267 int tx, int ty, int tw, int th,
268 const char *label,
269 const char *value, const char *unit,
270 AlertLevel level)
271{
272 COLORREF bg, fg;
273 char badge[24];
274 HPEN border_pen;
275 HPEN old_pen;
276 HBRUSH old_br;
277 char full_val[48];
278
279 switch (level) {
280 case ALERT_WARNING:
281 bg = CLR_WN_BG; fg = CLR_WN_FG;
282 snprintf(badge, sizeof(badge), " WARNING");
283 break;
284 case ALERT_CRITICAL:
285 bg = CLR_CR_BG; fg = CLR_CR_FG;
286 snprintf(badge, sizeof(badge), " CRITICAL");
287 break;
288 default:
289 bg = CLR_OK_BG; fg = CLR_OK_FG;
290 snprintf(badge, sizeof(badge), " NORMAL");
291 break;
292 }
293
294 /* Fill and border */
295 fill_rect(hdc, tx, ty, tw, th, bg);
296 border_pen = CreatePen(PS_SOLID, 2, fg);
297 old_pen = (HPEN) SelectObject(hdc, border_pen);
298 old_br = (HBRUSH)SelectObject(hdc, GetStockObject(NULL_BRUSH));
299 RoundRect(hdc, tx + 1, ty + 1, tx + tw - 1, ty + th - 1, 10, 10);
300 SelectObject(hdc, old_pen);
301 SelectObject(hdc, old_br);
302 DeleteObject(border_pen);
303
304 /* Parameter label (top-left) */
305 draw_text_ex(hdc, label, tx + 10, ty + 8, tw - 20, 18,
306 g_app.font_tile_lbl, fg,
307 DT_SINGLELINE | DT_LEFT);
308
309 /* Measurement value (centre) */
310 snprintf(full_val, sizeof(full_val), "%s %s", value, unit);
311 draw_text_ex(hdc, full_val, tx + 8, ty + 30, tw - 16, 36,
312 g_app.font_tile_val, CLR_DARK_TEXT,
313 DT_SINGLELINE | DT_LEFT | DT_VCENTER);
314
315 /* Status badge (bottom-left) */
316 draw_text_ex(hdc, badge, tx + 8, ty + th - 26, tw - 16, 20,
317 g_app.font_tile_lbl, fg,
318 DT_SINGLELINE | DT_LEFT);
319}

321/**
322 * @brief Paint the 2×2 vital sign tile grid.
323 * @req SWR-GUI-003
324 */
325static void paint_tiles(HDC hdc, int cw)
326{
327 const VitalSigns *v = NULL;
328 char hr_s[16], bp_s[24], tp_s[16], sp_s[16];
329 AlertLevel lv_hr = ALERT_NORMAL, lv_bp = ALERT_NORMAL;
330 AlertLevel lv_tp = ALERT_NORMAL, lv_sp = ALERT_NORMAL;
331
332 int pad = 10;
333 int tw = (cw - 3 * pad) / 2;
334 int th = (TILE_H - 3 * pad) / 2;
335
336 /* Background */
337 fill_rect(hdc, 0, TILE_Y, cw, TILE_H + pad, CLR_NEAR_WHITE);
338
339 if (g_app.has_patient) {
340 v = patient_latest_reading(&g_app.patient);
341 }
342
343 if (v != NULL) {
344 snprintf(hr_s, sizeof(hr_s), "%d", v->heart_rate);
345 snprintf(bp_s, sizeof(bp_s), "%d / %d", v->systolic_bp, v->diastolic_bp);
346 snprintf(tp_s, sizeof(tp_s), "%.1f", v->temperature);
347 snprintf(sp_s, sizeof(sp_s), "%d", v->spo2);
348 lv_hr = check_heart_rate(v->heart_rate);
349 lv_bp = check_blood_pressure(v->systolic_bp, v->diastolic_bp);
350 lv_tp = check_temperature(v->temperature);
351 lv_sp = check_spo2(v->spo2);
352 } else {
353 strncpy(hr_s, "--", sizeof(hr_s) - 1); hr_s[sizeof(hr_s) - 1] = '\0';
354 strncpy(bp_s, "--/--", sizeof(bp_s) - 1); bp_s[sizeof(bp_s) - 1] = '\0';
355 strncpy(tp_s, "--", sizeof(tp_s) - 1); tp_s[sizeof(tp_s) - 1] = '\0';
356 strncpy(sp_s, "--", sizeof(sp_s) - 1); sp_s[sizeof(sp_s) - 1] = '\0';
357 }
358
359 /* Row 1 */
360 paint_tile(hdc, pad, TILE_Y + pad, tw, th,
361 "HEART RATE", hr_s, "bpm", lv_hr);
362 paint_tile(hdc, pad + tw + pad, TILE_Y + pad, tw, th,
363 "BLOOD PRESSURE", bp_s, "mmHg", lv_bp);
364 /* Row 2 */
365 paint_tile(hdc, pad, TILE_Y + pad + th + pad, tw, th,
366 "TEMPERATURE", tp_s, "C", lv_tp);
367 paint_tile(hdc, pad + tw + pad, TILE_Y + pad + th + pad, tw, th,
368 "SpO2", sp_s, "%", lv_sp);
369}

503/**
504 * @brief Refresh all dynamic dashboard content after any data change.
505 * @req SWR-GUI-003
506 * @req SWR-GUI-004
507 */
508static void update_dashboard(HWND w)
509{
510 char buf[256];
511 const VitalSigns *latest = NULL;
512 Alert alerts[MAX_ALERTS];
513 int alert_count = 0;
514 int i;
515
516 SendMessageA(GetDlgItem(w, IDC_LIST_HISTORY), LB_RESETCONTENT, 0, 0);
517 SendMessageA(GetDlgItem(w, IDC_LIST_ALERTS), LB_RESETCONTENT, 0, 0);
518
519 /* Trigger repaint of the painted zones */
520 InvalidateRect(w, NULL, FALSE);
521
522 if (!g_app.has_patient) {
523 SendMessageA(GetDlgItem(w, IDC_LIST_ALERTS), LB_ADDSTRING, 0,
524 (LPARAM)"No patient admitted yet.");
525 return;
526 }
527
528 /* History list */
529 for (i = 0; i < g_app.patient.reading_count; ++i) {
530 const VitalSigns *r = &g_app.patient.readings[i];
531 snprintf(buf, sizeof(buf),
532 "#%d HR %d bpm | BP %d/%d mmHg | Temp %.1f C"
533 " | SpO2 %d%% [%s]",
534 i + 1,
535 r->heart_rate,
536 r->systolic_bp, r->diastolic_bp,
537 r->temperature,
538 r->spo2,
539 alert_level_str(overall_alert_level(r)));
540 SendMessageA(GetDlgItem(w, IDC_LIST_HISTORY), LB_ADDSTRING, 0, (LPARAM)buf);
541 }
542
543 /* Alerts list */
544 latest = patient_latest_reading(&g_app.patient);
545 if (latest != NULL) {
546 alert_count = generate_alerts(latest, alerts, MAX_ALERTS);
547 }
548
549 if (alert_count == 0) {
550 SendMessageA(GetDlgItem(w, IDC_LIST_ALERTS), LB_ADDSTRING, 0,
551 (LPARAM)"No active alerts — all parameters within normal range.");
552 } else {
553 for (i = 0; i < alert_count; ++i) {
554 const char *sev = (alerts[i].level == ALERT_CRITICAL)
555 ? "CRITICAL" : "WARNING ";
556 snprintf(buf, sizeof(buf), "[%s] %s", sev, alerts[i].message);
557 SendMessageA(GetDlgItem(w, IDC_LIST_ALERTS), LB_ADDSTRING, 0, (LPARAM)buf);
558 }
559 }
560}

578static int parse_int_field(HWND p, int id, const char *label, int *out)
579{
580 char buf[64]; char *ep = NULL; long v;
581 if (!get_txt(p, id, buf, (int)sizeof(buf))) {
582 char m[128]; snprintf(m, sizeof(m), "%s is required.", label);
583 MessageBoxA(p, m, APP_TITLE, MB_OK | MB_ICONWARNING);
584 SetFocus(GetDlgItem(p, id)); return 0;
585 }
586 v = strtol(buf, &ep, 10);
587 if (ep == buf || *ep != '\0') {
588 char m[128]; snprintf(m, sizeof(m), "%s must be a whole number.", label);
589 MessageBoxA(p, m, APP_TITLE, MB_OK | MB_ICONWARNING);
590 SetFocus(GetDlgItem(p, id)); return 0;
591 }
592 *out = (int)v; return 1;
593}
