3 * @brief Vital signs data types, thresholds, and validation functions.
4 *
5 * @details
6 * This module defines the core data structures and validation logic for
7 * patient vital signs. Each parameter is evaluated against clinically
8 * defined thresholds derived from AHA/ACC guidelines and classified into
9 * one of three alert levels: NORMAL, WARNING, or CRITICAL.
10 *
11 * ### Threshold Reference Table
12 * | Parameter | Normal | Warning | Critical |
13 * |----------------|-----------------|--------------------------|-------------------|
14 * | Heart Rate | 60–100 bpm | 41–59 / 101–150 bpm | ≤40 / ≥151 bpm |
15 * | Systolic BP | 90–140 mmHg | 71–89 / 141–180 mmHg | ≤70 / ≥181 mmHg |
16 * | Diastolic BP | 60–90 mmHg | 41–59 / 91–120 mmHg | ≤40 / ≥121 mmHg |
17 * | Temperature | 36.1–37.2 °C | 35.0–36.0 / 37.3–39.5 °C| <35.0 / >39.5 °C |
18 * | SpO2 | 95–100 % | 90–94 % | <90 % |
19 *
20 * ### IEC 62304 Traceability

39/**
40 * @brief Classification of a vital sign reading relative to clinical norms.
41 *
42 * @details The numeric ordering is intentional: ALERT_NORMAL < ALERT_WARNING
43 * < ALERT_CRITICAL, allowing comparisons such as `if (level > ALERT_NORMAL)`.
44 * This ordering is relied upon by overall_alert_level().
45 */
46typedef enum {
47 ALERT_NORMAL = 0, /**< Reading within normal clinical range. */
48 ALERT_WARNING = 1, /**< Reading outside normal range; clinician review
49 recommended. */
50 ALERT_CRITICAL = 2 /**< Reading at potentially life-threatening level;
51 immediate intervention required. */
52} AlertLevel;

58/**
59 * @brief A single set of patient vital sign measurements.
60 *
61 * @details All fields are captured at the same point in time and treated
62 * as a single observation. Callers must ensure values are physically
63 * plausible before passing them to validation functions.
64 */
65typedef struct {
66 int heart_rate; /**< Heart rate in beats per minute (bpm). */
67 int systolic_bp; /**< Systolic blood pressure in mmHg. */
68 int diastolic_bp; /**< Diastolic blood pressure in mmHg. */
69 float temperature; /**< Core body temperature in degrees Celsius (°C). */
70 int spo2; /**< Peripheral oxygen saturation as a percentage. */
71} VitalSigns;

77/**
78 * @brief Classify a heart rate reading against clinical thresholds.
79 *
80 * @details
81 * - **NORMAL**: 60–100 bpm (inclusive)\n
82 * - **WARNING**: 41–59 bpm or 101–150 bpm\n
83 * - **CRITICAL**: ≤40 bpm or ≥151 bpm
84 *
85 * @param[in] bpm Heart rate in beats per minute. Negative values are
86 * treated as CRITICAL.
87 * @return AlertLevel classification for the given heart rate.
88 *
89 * @par Requirement
90 * SWR-VIT-001
91 */
92AlertLevel check_heart_rate(int bpm);
93
94/**
95 * @brief Classify a blood pressure reading against clinical thresholds.
96 *
97 * @details Both systolic and diastolic values are evaluated independently.
98 * The more severe of the two classifications is returned.
99 *
100 * - **NORMAL**: Systolic 90–140 mmHg AND diastolic 60–90 mmHg\n
101 * - **WARNING**: Systolic 71–89 / 141–180 mmHg OR diastolic 41–59 / 91–120 mmHg\n
102 * - **CRITICAL**: Systolic ≤70 / ≥181 mmHg OR diastolic ≤40 / ≥121 mmHg
103 *
104 * @param[in] systolic Systolic pressure in mmHg.
105 * @param[in] diastolic Diastolic pressure in mmHg.
106 * @return AlertLevel classification for the combined blood pressure reading.
107 *
108 * @par Requirement
109 * SWR-VIT-002
110 */
111AlertLevel check_blood_pressure(int systolic, int diastolic);
112
113/**
114 * @brief Classify a body temperature reading against clinical thresholds.
115 *
116 * @details
117 * - **NORMAL**: 36.1–37.2 °C (inclusive)\n
118 * - **WARNING**: 35.0–36.0 °C or 37.3–39.5 °C\n
119 * - **CRITICAL**: <35.0 °C (hypothermia) or >39.5 °C (hyperpyrexia)
120 *
121 * @param[in] temp_c Body temperature in degrees Celsius.
122 * @return AlertLevel classification for the given temperature.
123 *
124 * @par Requirement
125 * SWR-VIT-003
126 */
127AlertLevel check_temperature(float temp_c);
128
129/**
130 * @brief Classify a peripheral oxygen saturation reading.
131 *
132 * @details
133 * - **NORMAL**: 95–100 % (inclusive)\n
134 * - **WARNING**: 90–94 %\n
135 * - **CRITICAL**: <90 % (hypoxaemia)
136 *
137 * @param[in] spo2 SpO2 as a percentage (0–100). Values outside this range
138 * are treated as CRITICAL.
139 * @return AlertLevel classification for the given SpO2 reading.
140 *
141 * @par Requirement
142 * SWR-VIT-004
143 */
144AlertLevel check_spo2(int spo2);

150/**
151 * @brief Return the highest alert level across all vital sign parameters.
152 *
153 * @details Internally calls check_heart_rate(), check_blood_pressure(),
154 * check_temperature(), and check_spo2(), then returns the maximum value.
155 * This ensures that any single critical parameter elevates the overall
156 * patient status to CRITICAL regardless of other parameters.
157 *
158 * @param[in] vitals Pointer to a valid VitalSigns structure. Must not be NULL.
159 * @return The highest AlertLevel among all four parameters.
160 *
161 * @pre vitals != NULL
162 * @par Requirement
163 * SWR-VIT-005
164 */
165AlertLevel overall_alert_level(const VitalSigns *vitals);
