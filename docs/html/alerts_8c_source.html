31/**
32 * @brief Generate structured alert records for out-of-range vital signs.
33 *
34 * @details
35 * Internal flow for each parameter:
36 * -# Call the corresponding check_*() function.
37 * -# If the result is not ALERT_NORMAL, write an Alert record via PUSH_ALERT.
38 * -# PUSH_ALERT enforces the max_out cap before writing.
39 *
40 * The PUSH_ALERT macro copies the parameter name with strncpy (max 31 chars
41 * + null) and formats the message with snprintf (max ALERT_MSG_LEN - 1 chars
42 * + null), guaranteeing null-termination in both fields.
43 */
44int generate_alerts(const VitalSigns *vitals, Alert *out, int max_out)
45{
46 int count = 0;
47 AlertLevel lvl;
48
49 /* Write one Alert record if count < max_out. */
50#define PUSH_ALERT(_al, param, fmt, ...) \
51 do { \
52 if (count < max_out) { \
53 out[count].level = (_al); \
54 strncpy(out[count].parameter, (param), 31); \
55 out[count].parameter[31] = '\0'; \
56 snprintf(out[count].message, ALERT_MSG_LEN, fmt, __VA_ARGS__); \
57 count++; \
58 } \
59 } while (0)
60
61 lvl = check_heart_rate(vitals->heart_rate);
62 if (lvl != ALERT_NORMAL)
63 PUSH_ALERT(lvl, "Heart Rate",
64 "Heart rate %d bpm [normal 60-100]",
65 vitals->heart_rate);
66
67 lvl = check_blood_pressure(vitals->systolic_bp, vitals->diastolic_bp);
68 if (lvl != ALERT_NORMAL)
69 PUSH_ALERT(lvl, "Blood Pressure",
70 "BP %d/%d mmHg [normal 90-140 / 60-90]",
71 vitals->systolic_bp, vitals->diastolic_bp);
72
73 lvl = check_temperature(vitals->temperature);
74 if (lvl != ALERT_NORMAL)
75 PUSH_ALERT(lvl, "Temperature",
76 "Temp %.1f C [normal 36.1-37.2]",
77 (double)vitals->temperature);
78
79 lvl = check_spo2(vitals->spo2);
80 if (lvl != ALERT_NORMAL)
81 PUSH_ALERT(lvl, "SpO2",
82 "SpO2 %d%% [normal 95-100%%]",
83 vitals->spo2);
84
85#undef PUSH_ALERT
86 return count;
87}
